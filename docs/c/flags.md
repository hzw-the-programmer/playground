### 如何使用
```
#define FLAG_1 0x01 // 0000 0001
#define FLAG_2 0x02 // 0000 0010
#define FLAG_3 0x04 // 0000 0100
#define FLAG_4 0x08 // 0000 1000
#define FLAG_5 0x10 // 0001 0000

uint8_t flags = 0;

// 开 FLAG_1
flags |= FLAG_1; // 16进制：0x01，2进制：0000 0001
// 开 FLAG_2
flags |= FLAG_2; // 16进制：0x03，2进制：0000 0011

// 检测 FLAG_1 是否开启
flags & FALG_1 // 表达式 true
// 检测 FLAG_2 是否开启
flags & FALG_2 // 表达式 true
// 检测 FLAG_1，FLAG_2 至少一个开启
flags & (FLAG_1 | FLAG_2)
// 检测 FLAG_1，FLAG_2 同时开启
flags & (FLAG_1 | FALG_2) == FLAG_1 | FLAG_2

// 关 FLAG_1
flags &= ~FLAG_1; // 16进制：0x02，2进制：0000 0010
// 关 FALG_2
flags &= ~FLAG_2; // 16进制：0x00，2进制：0000 0000
```

### 开
```
#define FLAG_1 0x01 // 0000 0001
#define FLAG_2 0x02 // 0000 0010

uint8_t flags = 0;

flags |= FLAG_1;
// 16进制：0x01，2进制：0000 0001

flags |= FLAG_2;
// 16进制：0x03，2进制：0000 0011
```

### 关
```
#define FLAG_1 0x01 // 0000 0001
#define FLAG_2 0x02 // 0000 0010

uint8_t flags = FLAG_1 | FLAG_2;
// 16进制：0x03，2进制：0000 0011

flags &= ~FLAG_1;
// ~FLAG_1 16进制：0xfe，2进制：1111 1110
// flags 16进制：0x02，2进制：0000 0010

flags &= ~FLAG_2;
// ~FLAG_2 16进制：0xfd，2进制：1111 1101
// flags 16进制：0x00，2进制：0000 0000
```

### 8421法
```
8 == 1000
4 == 0100
2 == 0010
1 == 0001
```

### 8421组合
```
A = 8 + 2 == 1000 + 0010 = 1010
B = 8 + 2 + 1 == 1000 + 0010 + 0001 = 1011
C = 8 + 4 == 1000 + 0100 = 1100
D = 8 + 4 + 1 == 1000 + 0100 + 0001 = 1101
E = 8 + 4 + 2 == 1000 + 0100 + 0010 = 1110
F = 8 + 4 + 2 + 1 == 1000 + 0100 + 0010 + 0001 = 1111
```
