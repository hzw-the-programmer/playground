typedef struct {
    void *ptr;
    kal_int32 mcount;
    kal_int32 fcount;
} Node;

#define PTRS_CAP (10000)
static Node ptrs[PTRS_CAP];
static kal_int32 ptrs_len;

static kal_int32 mcount;
static kal_int32 fcount;

static void* optrs[9];

void* sslmalloc(kal_int32 mem_size, void *user_data) {
    kal_int32 i;
    void *ptr = MALLOC(mem_size);

    mcount++;

    for (i = 0; i < ptrs_len; i++) {
        if (ptrs[i].ptr == ptr) {
            ptrs[i].mcount++;
            return ptr;
        }
    }

    ptrs[i].ptr = ptr;
    ptrs[i].mcount++;
    ptrs_len++;

    return ptr;
}

void sslfree(void *mem_ptr, void *user_data) {
    kal_int32 i;

    fcount++;
    for (i = 0; i < ptrs_len; i++) {
        if (ptrs[i].ptr == mem_ptr) {
            ptrs[i].fcount++;
        }
    }
    FREE(mem_ptr);
}
